#include <FastLED.h>

#define NUM_LEDS 33  // Number of LEDs in the strip
#define DATA_PIN 6   // Pin for LED data line

CRGB leds[NUM_LEDS];

void setup() {
  FastLED.addLeds<WS2812B, DATA_PIN, GRB>(leds, NUM_LEDS);
  FastLED.setBrightness(200);
  fill_solid(leds, NUM_LEDS, CRGB::Black);
  FastLED.show();
}

void loop() {
  pumpEffect();
}

void pumpEffect() {
  for (int i = 0; i < NUM_LEDS; i++) {  
    // Set the current LED color (red to blue blend)
    leds[i] = blend(CRGB(255, 0, 0), CRGB(0, 0, 255), map(i, 0, NUM_LEDS - 1, 0, 255));

    // Gradually fade previous LEDs
    for (int j = 0; j < NUM_LEDS; j++) {
      leds[j].fadeToBlackBy(80);  // Lower = longer fade effect
    }

    FastLED.show();

    // ðŸ”¥ Increased exponent for **slower start + faster acceleration**
    float delayFactor = pow((1.0f - ((float)i / (NUM_LEDS - 1))), 4.5);  
    int delayTime = 2 + (120 * delayFactor);  // Higher max delay for a more dramatic ramp-up

    delay(delayTime);
  }
}
